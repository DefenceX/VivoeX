include(FindPkgConfig)
project(hmi-display)
cmake_minimum_required(VERSION 2.8)
add_executable(hmi-display  main.cpp rendererSdl.cpp rendererCairo.cpp rendererGva.cpp screenGva.cpp ./common/debug.cpp ./common/logGva.cpp ./common/pngget.cpp)
#add_subdirectory(dds)
#add_subdirectory(display)
add_subdirectory(common)
set (CMAKE_CXX_FLAGS "-w")
pkg_check_modules(CAIRO REQUIRED cairo)
target_link_libraries(hmi-display SDL2 X11 png pthread)
target_link_libraries(hmi-display ${CAIRO_LIBRARIES})
target_include_directories(hmi-display PUBLIC ${CAIRO_INCLUDE_DIRS} common dds display)
target_compile_options(hmi-display PUBLIC ${CAIRO_CFLAGS_OTHER})
install(TARGETS hmi-display DESTINATION bin)
